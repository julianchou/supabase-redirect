import 'package:flutter/material.dart'; import 'package:supabase_flutter/supabase_flutter.dart'; import 'package:app_links/app_links.dart'; class InitPage extends StatefulWidget { const InitPage({super.key}); @override State createState() => _InitPageState(); } class _InitPageState extends State { final _appLinks = AppLinks(); bool _hasHandled = false; @override void initState() { super.initState(); _listenToDeepLinks(); _checkAndInit(); } void _listenToDeepLinks() { _appLinks.uriLinkStream.listen((uri) { if (_hasHandled || uri == null) return; final deeplink = uri.toString(); print("📩 收到 Deep Link：$deeplink"); if (uri.scheme == 'io.supabase.flutter' && uri.host == 'login-callback') { _hasHandled = true; _handleLoginRedirect(deeplink); } }); } Future _handleLoginRedirect(String deeplink) async { try { final response = await Supabase.instance.client.auth.exchangeCodeForSession(deeplink); final session = response.session; if (session != null) { print("✅ 登入成功：${session.user?.email}"); if (mounted) { Navigator.of(context).pushNamedAndRemoveUntil('/home', (route) => false); } } else { _showError("⚠️ 登入失敗，請重新嘗試"); } } catch (e) { print("❗️ exchangeCodeForSession 發生錯誤：$e"); _showError("驗證失敗，請重新登入"); } } void _showError(String message) { if (!mounted) return; ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text(message))); Navigator.of(context).pushReplacementNamed('/login'); } Future _checkAndInit() async { final user = Supabase.instance.client.auth.currentUser; print("🔍 檢查目前使用者：${user?.email}"); if (user == null) { Navigator.of(context).pushReplacementNamed('/login'); return; } final uid = user.id; final profile = await Supabase.instance.client .from('profiles') .select() .eq('id', uid) .maybeSingle(); if (profile == null) { await Supabase.instance.client.from('profiles').insert({ 'id': uid, 'nickname': user.email?.split('@')[0], }); } Navigator.of(context).pushReplacementNamed('/home'); } @override Widget build(BuildContext context) { return const Scaffold(body: Center(child: CircularProgressIndicator())); } }
